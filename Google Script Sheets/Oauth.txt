const SF_LOGIN_URL = 'https://login.salesforce.com'; // or https://test.salesforce.com

function getSalesforceService() {
  const props = PropertiesService.getScriptProperties();
  const clientId = props.getProperty('Consumer Key');
  const clientSecret = props.getProperty('Consumer Secret');
  if (!clientId || !clientSecret) throw new Error('Missing Consumer Key/Secret in Script Properties.');

  return OAuth2.createService('salesforce')
    .setAuthorizationBaseUrl(SF_LOGIN_URL + '/services/oauth2/authorize')
    .setTokenUrl(SF_LOGIN_URL + '/services/oauth2/token')
    .setClientId(clientId)
    .setClientSecret(clientSecret)
    .setCallbackFunction('authCallback')
    .setPropertyStore(PropertiesService.getUserProperties())
    .setScope('api refresh_token offline_access')
    .setParam('prompt', 'consent');
}

function authCallback(request) {
  const service = getSalesforceService();
  const authorized = service.handleCallback(request);
  return HtmlService.createHtmlOutput(
    authorized ? 'Salesforce authorization successful. You can close this tab.'
               : 'Salesforce authorization denied.'
  );
}

function authorizeSalesforce() {
  const service = getSalesforceService();
  if (service.hasAccess()) {
    SpreadsheetApp.getUi().alert('Salesforce: Already authorized ✓');
    return 'Already authorized';
  }

  const url = service.getAuthorizationUrl();
  showAuthUrlDialog_(url);
  return url;
}

function checkAuth() {
  const authorized = getSalesforceService().hasAccess();
  if (authorized) {
    SpreadsheetApp.getUi().alert('Salesforce: Authorized ✓');
  } else {
    reAuthorizeSalesforce();
  }
}

function reAuthorizeSalesforce() {
  getSalesforceService().reset();
  authorizeSalesforce(); // clears token then immediately shows auth URL dialog
}

function forceReauthSalesforce() {
  reAuthorizeSalesforce(); // back-compat alias
}

function showAuthUrlDialog_(url) {
  const html = HtmlService.createHtmlOutput(
    '<p style="font-family:sans-serif">Click below to authorize Salesforce, ' +
    'then close this dialog and run Sync again.</p>' +
    '<p><a href="' + url + '" target="_blank" style="font-size:14px">' +
    'Authorize Salesforce →</a></p>'
  ).setWidth(450).setHeight(120);
  SpreadsheetApp.getUi().showModalDialog(html, 'Salesforce Authorization Required');
}
